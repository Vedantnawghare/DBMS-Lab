#to print salary and name with emp_id and excluding the exceptions=> 
create or replace procedure excl(n in number)
as
    vname    employees.emp_name%type;
    vsalary  employees.salary%type;
begin
    select emp_name,salary
    into vname, vsalary
    from employees
    where emp_id = n;

    DBMS_OUTPUT.PUT_LINE('Emp Name = ' || vname);
    DBMS_OUTPUT.PUT_LINE('Emp Salary = ' || vsalary);

exception
    when no_data_found then
        DBMS_OUTPUT.PUT_LINE('Emp ID not found');
end;
/
to run
 set serveroutput on;
exec excl(101);
