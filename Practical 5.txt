1) create table departments (dept_id NUMBER PRIMARY KEY,dept_name VARCHAR2(50),location VARCHAR2(50) );


2) create table employees (emp_id NUMBER PRIMARY KEY, emp_name VARCHAR2(50), dept_id NUMBER, salary NUMBER(10,2), manager_id NUMBER, CONSTRAINT fk_emp_dept FOREIGN KEY (dept_id) REFERENCES departments(dept_id), CONSTRAINT fk_emp_manager FOREIGN KEY (manager_id) REFERENCES employees(emp_id) );


3) create table projects (proj_id NUMBER PRIMARY KEY, proj_name VARCHAR2(50), dept_id NUMBER, CONSTRAINT fk_proj_dept FOREIGN KEY (dept_id) REFERENCES departments(dept_id) );


4) create table assignments ( emp_id NUMBER, proj_id NUMBER, hours_worked NUMBER, PRIMARY KEY (emp_id, proj_id), CONSTRAINT fk_assign_emp FOREIGN KEY (emp_id) REFERENCES employees(emp_id), CONSTRAINT fk_assign_proj FOREIGN KEY (proj_id) REFERENCES projects(proj_id) );
